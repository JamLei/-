  • gitea 配置

    version: "3"

networks:
    gitea:
        external: false

services:

    ...

    gitea-server:
        image: gitea/gitea:1.14.1
        container_name: gitea
        # restart: always
        networks:
            - gitea
        environment:
            - USER_UID=1000
            - USER_GID=1000
            - GITEA__database__DB_TYPE=mysql
            - GITEA__database__HOST=gitea-db:3306
            - GITEA__database__NAME=gitea
            - GITEA__database__USER=gitea
            - GITEA__database__PASSWD=gitea
        volumes:
            - /home/docker/gitea:/data
            - /etc/timezone:/etc/timezone:ro
            - /etc/localtime:/etc/localtime:ro
        ports:
            - "3000:3000"
            - "222:22"
        depends_on:
            - gitea-db

    上述配置有几处注意的地方：

    1. GITEA__database__HOST=gitea-db:3306 中的 gitea-db 是 gitea 所依赖的 mysql 容器的名字，而 3306 是它的端口。这和上面的 mysql 配置是遥相呼应的。

    2. gitea 映射了宿主机的端口，特别是 ssh 端口，未来在使用中，我们配置、使用的应该是宿主的 IP 和端口。
